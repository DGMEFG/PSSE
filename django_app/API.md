# USER
## 发送验证码
    /api/send_verification_code
### INPUT
    {
        "email": "admin@example.com"
    }
### OUTPUT
    {
        "success": True
    }
## 登录
    /api/login
### INPUT
    {
        "email": "admin@example.com",
        "password": "admin"
    }
### OUTPUT
    {
        "success": True,
        "message": "登录成功" 200 / "用户名或密码错误" 401
    }

## 注册
    /api/register
### INPUT
    {
        "email": "admin@example.com",
        "username": "admin"
        "password": "admin"
        "code": "123456"
    }
### OUTPUT
    {
        "success": True,
        "message": "注册成功" 201 / "验证码错误" 401 / "邮箱已被注册"400 / "请先获取验证码" 400 / "验证码已过期，请重新获取" 401
    }
## 修改密码
    /api/change_password
### INPUT
    {
        "code": "123456",
        "old_password": "admin",
        "new_password": "admin1"
    }
### OUTPUT
    {
        "success": True,
        "message": "修改成功" / "验证码不正确或已过期。" / "旧密码不正确。"
    }
